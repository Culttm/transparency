Context stores the original template elements and is responsible for creating, adding and removing template instances to match the amount of models.

 
class Context

  detach = chainable ->
    @parent = @el.parentNode
    if @parent
      @nextSibling = @el.nextSibling
      @parent.removeChild @el

  attach = chainable ->
    if @parent
      if @nextSibling
      then @parent.insertBefore @el, @nextSibling
      else @parent.appendChild @el

  constructor: (@el) ->
    @template      = cloneNode @el
    @instances     = [new Instance(@el)]
